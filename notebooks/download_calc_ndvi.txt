example #1: add two files together
gdal_calc.py -A input1.tif -B input2.tif --outfile=result.tif --calc="A+B"

example #2: average of two files
gdal_calc.py -A input.tif -B input2.tif --outfile=result.tif --calc="(A+B)/2"

example #3: average of many files
gdal_calc.py -A input.tif input2.tif input3.tif ... inputN.tif --outfile=result.tif --calc="average(A,axis=0)"

example #4: average of many files (using wildcards)
gdal_calc.py -A input*.tif --outfile=result.tif --calc="average(A,axis=0)"

example #5: set values of zero and below to null
gdal_calc.py -A input.tif --outfile=result.tif --calc="A*(A>0)" --NoDataValue=0

example #6: using logical operator to keep a range of values from input
gdal_calc.py -A input.tif --outfile=result.tif --calc="A*logical_and(A>100,A<150)"

example #7: work with multiple bands
